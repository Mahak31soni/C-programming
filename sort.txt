Using Hash Map(PY):
arr = [4,1,7,6,8]
res = []
length = len(arr)
hash= [0] * 10

for i in range(length):
	hash[arr[i]] = 1
print(hash)
for i in range(len(hash)):
    if hash[i] == 1:
        res.append(i)
print(res)

arr = [4,1,7,6,8]
length = len(arr)
for i in range(length - 1):
	for j in range(i+1, length):
		if arr[i] > arr[j]:
			arr[i], arr[j] = arr[j], arr[i]
